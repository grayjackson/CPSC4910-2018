CREATE TABLE DRIVERS(
	ID INTEGER NOT NULL AUTO_INCREMENT,
	FNAME VARCHAR(25) NOT NULL,
	LNAME VARCHAR(25) NOT NULL,
	PHONE VARCHAR(15),
	EMAIL VARCHAR(25) NOT NULL,
	STREET_ADDRESS VARCHAR(100),
	CITY VARCHAR(50),
	STATE_CODE VARCHAR(5),
	ZIP INTEGER,
	POINTS INTEGER,
	SPONSOR_ID INTEGER NOT NULL,
	PRIMARY KEY (ID)
);

CREATE TABLE SPONSORS(
	ID INTEGER NOT NULL AUTO_INCREMENT,
	NAME VARCHAR(200) NOT NULL,
	PHONE VARCHAR(15),
	EMAIL VARCHAR(25) NOT NULL,
	WEBSITE VARCHAR(100),
	STREET_ADDRESS VARCHAR(100),
	CITY VARCHAR(50),
	STATE_CODE VARCHAR(5),
	ZIP INTEGER,
	POINT_VALUE DECIMAL (5,2),
	PRIMARY KEY (ID)
);

CREATE TABLE ADMINS(
	ID INTEGER NOT NULL AUTO_INCREMENT,
	FNAME VARCHAR(25) NOT NULL,
	LNAME VARCHAR(25) NOT NULL,
	EMAIL VARCHAR(25) NOT NULL,
	PRIMARY KEY (ID)
);

CREATE TABLE INACTIVE_DRIVERS(
	ID INTEGER NOT NULL,
	FNAME VARCHAR(25) NOT NULL,
	LNAME VARCHAR(25) NOT NULL,
	PHONE VARCHAR(15),
	EMAIL VARCHAR(25) NOT NULL,
	ADDRESS VARCHAR(100) NOT NULL,
	POINTS INTEGER,
	SPONSOR_ID INTEGER NOT NULL,
	PRIMARY KEY (ID)
);

CREATE TABLE INACTIVE_SPONSORS(
	ID INTEGER NOT NULL,
	NAME VARCHAR(50) NOT NULL,
	PHONE VARCHAR(15),
	EMAIL VARCHAR(25) NOT NULL,
	WEBSITE VARCHAR(100),
	ADDRESS VARCHAR(100),
	POINT_VALUE DECIMAL (5,2),
	PRIMARY KEY (ID)
);

CREATE TABLE INACTIVE_ADMINS(
	ID INTEGER NOT NULL,
	FNAME VARCHAR(25) NOT NULL,
	LNAME VARCHAR(25) NOT NULL,
	EMAIL VARCHAR(25) NOT NULL,
	PRIMARY KEY (ID)
);

CREATE TABLE DRIVER_LOGIN(
	DRIVERID INTEGER NOT NULL,
	UNAME VARCHAR(25) NOT NULL,
	PWORD VARCHAR(25) NOT NULL,
	PRIMARY KEY (DRIVERID)
);

CREATE TABLE SPONSOR_LOGIN(
	COID INTEGER NOT NULL,
	UNAME VARCHAR(25) NOT NULL,
	PWORD VARCHAR(25) NOT NULL,
	PRIMARY KEY (COID)
);

CREATE TABLE ADMIN_LOGIN(
	ADMINID INTEGER NOT NULL,
	UNAME VARCHAR(25) NOT NULL,
	PWORD VARCHAR(25) NOT NULL,
	PRIMARY KEY (ADMINID)
);

CREATE TABLE PURCHASES(
	PID INTEGER NOT NULL AUTO_INCREMENT,
	DRIVERID INTEGER NOT NULL,
	COID INTEGER NOT NULL,
	AMT_DOLLARS DECIMAL(5,2),
	AMT_POINTS INTEGER,
	PROFIT DECIMAL (5,2),
	PRIMARY KEY (PID)
);



CREATE TABLE PURCHASE_HISTORY(
	DRIVERID INTEGER NOT NULL,
	PURCHASEID INTEGER NOT NULL,
	PURCHASE_POINT_AMT INTEGER,
	TOTAL_POINTS_LEFT INTEGER,
	PRIMARY KEY (DRIVERID, PURCHASEID)
);
	
CREATE TABLE DRIVER_POINT_HISTORY(
	DRIVERID INTEGER NOT NULL,
	COID INTEGER NOT NULL,
	POINT_AMT INTEGER,
	TRANS_TIME TIMESTAMP,
	PRIMARY KEY(DRIVERID,COID)
);
	
CREATE TABLE MONTHLY_TOTAL_PROFIT(
	ID INTEGER NOT NULL,
	MONTH VARCHAR(20),
	YEAR INTEGER,
	TOTAL_MONTH_PROFIT DECIMAL(10,2),
	PRIMARY KEY (ID)
);
	
CREATE TABLE TOTAL_PROFIT(
	AMT_PROFIT DECIMAL(15,2)
);

CREATE TABLE HELP_DESK(
	ID INTEGER NOT NULL AUTO_INCREMENT,
	ISSUE VARCHAR(8000),
	PRIMARY KEY (ID)
);
